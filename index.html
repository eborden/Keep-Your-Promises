Single Use Workers

One Time Worker


promiseWorker = (script, message) => {
    var d = $.Deferred(),
        w = new Worker(script);

    w.onmessage = e => d.resolve(e.data);
    w.onerror = e => d.reject(e);

    d.always(() => w.terminate());

    w.postMessage(message);
    return d.promise();
}

//One use worker

    .then(data => console.log(data));
